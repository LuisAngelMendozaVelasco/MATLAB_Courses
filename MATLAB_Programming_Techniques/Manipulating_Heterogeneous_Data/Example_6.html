Set up Activity

function tr = loadCrashTestData
load ./data/crashData.mat
 
testObjectives = [crashData.testInfo]';
impactSpeeds = [crashData.impactSpeed]';
sizes = cat(1,crashData.carSize);
weight = sizes(:,1);
 
allData = cat(1,crashData.data);
numericData = allData(:,[2 4 6]);
 
ma = cellfun(@(x) max(abs(x)),numericData);
 
tr = table(testObjectives,impactSpeeds,weight, ...
ma(:,1),ma(:,2),ma(:,3),...
'VariableNames',["Test Type" "Impact Speed" "Weight" "Max X" "Max Y" "Max Z"]);
end
