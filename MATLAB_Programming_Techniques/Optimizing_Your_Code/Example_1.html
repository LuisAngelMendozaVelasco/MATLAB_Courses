function [Fmax, maxIdx, Fmean] = calcF(sensors)
n = size(sensors,1);
for k = 1:n
s = trimCell(sensors(k,:));
Fx = s{1}(:,2);
Fy = s{2}(:,2);
Fz = s{3}(:,2);
m = length(Fx);
for kk = 1:m
F(kk) = sqrt(Fx(kk)^2 + Fy(kk)^2 + Fz(kk)^2);
end
[Fmax(k), maxIdx(k)] = max(F);
Fmean(k) = mean(F);
end
end
 
function F = trimCell(F)
% A helper function to trim a cell array of column vectors to the
% so all the column vectors have the same number of rows. It will
% also work for matrices provided they have more rows than columns.
% It will return the result as a cell array.
l = cellfun(@length, F);
minL = min(l);
f = @(x) x(1:minL,:);
F = cellfun(f, F, "UniformOutput", false);
end
