Positive roots are real

function posRootTest(testcase)
x = linspace(0,pi);
y = sqrt(x);
verifyTrue(testcase,isreal(y))
end

Negative roots are complex

function negRootTest(testcase)
z = linspace(-pi,pi);
y = sqrt(z);
verifyFalse(testcase,isreal(y))
end

sqrt(x) is smaller than x (if x > 1)

function sqrtTest(testcase)
x = linspace(0,pi);
y = x + 2;
z = sqrt(y);
verifyGreaterThan(testcase,y,z)
end
